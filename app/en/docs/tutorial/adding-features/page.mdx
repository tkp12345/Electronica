## 1) Application menu

```js filename="main.js"
const { app, Menu } = require('electron')

const template = [
  {
    label: 'File',
    submenu: [
      { label: 'New', accelerator: 'CmdOrCtrl+N', click: () => console.log('new') },
      { type: 'separator' },
      { role: 'quit' },
    ],
  },
]

Menu.setApplicationMenu(Menu.buildFromTemplate(template))
```

<Callout type="info" emoji="🧒">
`accelerator: 'CmdOrCtrl+N'` is shorthand for _"⌘N on macOS, Ctrl+N everywhere else."_ Electron handles the per-OS mapping for you.
</Callout>

## 2) Global shortcut (works even when the app is in the background)

```js filename="main.js"
const { app, globalShortcut } = require('electron')

app.whenReady().then(() => {
  globalShortcut.register('CommandOrControl+Shift+P', () => {
    console.log('Palette!')
  })
})

app.on('will-quit', () => globalShortcut.unregisterAll())
```

## 3) Dialogs

```js filename="main.js"
const { dialog } = require('electron')

dialog.showOpenDialog({
  properties: ['openFile'],
  filters: [{ name: 'Images', extensions: ['png', 'jpg'] }],
}).then(({ filePaths }) => console.log(filePaths))
```

## 4) System notifications

```js filename="renderer or preload"
new Notification('Build finished', { body: 'Took 5.2s' })
```

<Callout type="info" emoji="🧒">
Same shape as the browser Notification API — Electron just routes it to the real OS notification center.
</Callout>

## 5) Tray icon

```js filename="main.js"
const { app, Tray, Menu } = require('electron')
const path = require('node:path')

let tray
app.whenReady().then(() => {
  tray = new Tray(path.join(__dirname, 'icon.png'))
  tray.setToolTip('My App')
  tray.setContextMenu(Menu.buildFromTemplate([{ role: 'quit' }]))
})
```
